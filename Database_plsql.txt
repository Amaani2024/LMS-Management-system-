CREATE OR REPLACE PROCEDURE generate_student_credentials(
    p_student_id   IN  NUMBER,
    p_password_out OUT VARCHAR2
) AS
    v_password VARCHAR2(20);
BEGIN
    v_password := 'TRN@' || LPAD(p_student_id, 4, '0');
    
    UPDATE STUDENTS
    SET STATUS         = 'ACTIVE',
        LOGIN_STATUS   = 'ACTIVE',
        GENERATED_PASS = v_password
    WHERE STUDENT_ID   = p_student_id;
    
    UPDATE ENTROLLMENT 
    SET STATUS = 'ACTIVE'
    WHERE STUDENT_ID = p_student_id;
    
    COMMIT;
    p_password_out := v_password;
EXCEPTION
    WHEN OTHERS THEN ROLLBACK; RAISE;
END generate_student_credentials;
/

CREATE OR REPLACE PROCEDURE record_payment_and_check(
    p_student_id    IN  NUMBER,
    p_enroll_id     IN  NUMBER,
    p_amount_paid   IN  NUMBER,
    p_total_amount  IN  NUMBER,
    p_method        IN  VARCHAR2,
    p_txn_ref       IN  VARCHAR2,
    p_pay_type      IN  VARCHAR2,
    p_auto_approved OUT VARCHAR2,
    p_password_out  OUT VARCHAR2
) AS
    v_balance    NUMBER;
    v_min        NUMBER;
    v_pay_status VARCHAR2(20);
BEGIN
    v_balance    := p_total_amount - p_amount_paid;
    v_min        := p_total_amount * 0.5;
    v_pay_status := CASE WHEN v_balance <= 0 THEN 'Paid' ELSE 'Partial' END;

    INSERT INTO PAYMENT (
        PAYMENT_ID, STUDENT_ID, ENTROLLMENT_ID,
        AMOUNT, TOTAL_AMOUNT, PAID_AMOUNT, BALANCE,
        PAYMENT_TYPE, PAYMENT_METHOD, PAYMENT_DATE,
        TRANSACTION_REF, STATUS
    ) VALUES (
        SEQ_PAYMENT.NEXTVAL, p_student_id, p_enroll_id,
        p_amount_paid, p_total_amount, p_amount_paid, v_balance,
        p_pay_type, p_method, SYSDATE,
        p_txn_ref, v_pay_status
    );
    COMMIT;

    IF p_amount_paid >= v_min THEN
        generate_student_credentials(p_student_id, p_password_out);
        UPDATE STUDENTS
        SET PAYMENT_STATUS = v_pay_status
        WHERE STUDENT_ID = p_student_id;
        COMMIT;
        p_auto_approved := 'YES';
    ELSE
        p_auto_approved := 'NO';
        p_password_out  := NULL;
    END IF;
EXCEPTION
    WHEN OTHERS THEN ROLLBACK; RAISE;
END record_payment_and_check;
/

-- Fix SHOW_STUDENTS_BY_COURSE
CREATE OR REPLACE PROCEDURE show_students_by_course(
    p_course_id IN NUMBER
) AS
    CURSOR c_course_students IS
        SELECT s.student_id, s.f_name, s.l_name, c.course_name
        FROM students s
        JOIN entrollment e ON s.student_id = e.student_id
        JOIN course c ON e.course_id = c.course_id
        WHERE e.course_id = p_course_id;
    v_rec c_course_students%ROWTYPE;
BEGIN
    OPEN c_course_students;
    LOOP
        FETCH c_course_students INTO v_rec;
        EXIT WHEN c_course_students%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(
            'ID: ' || v_rec.student_id ||
            ' Name: ' || v_rec.f_name || ' ' || v_rec.l_name ||
            ' Course: ' || v_rec.course_name
        );
    END LOOP;
    CLOSE c_course_students;
END show_students_by_course;
